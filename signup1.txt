document.getElementById('signupForm').addEventListener('submit', async function(e) {
    e.preventDefault();
    console.log('Form submitted');
    
    // Get form values
    const name = document.getElementById('name').value.trim();
    const surname = document.getElementById('surname').value.trim();
    const walletId = document.getElementById('walletId').value.trim();
    const pictureFile = document.getElementById('picture').files[0];

    // Validate inputs
    if (!name || !surname || !walletId || !pictureFile) {
        alert('Please fill in all fields');
        return;
    }

    // Prepare form data
    const formData = new FormData();
    formData.append('name', name);
    formData.append('surname', surname);
    formData.append('walletId', walletId);
    formData.append('picture', pictureFile);

    try {
        console.log('Sending request to server...');
        const response = await fetch('http://localhost:3000/signup', {
            method: 'POST',
            body: formData
        });

        if (!response.ok) {
            const errorData = await response.json();
            throw new Error(errorData.error || 'Server error');
        }

        const data = await response.json();
        console.log('Received response:', data);

        // Parse and display the SVG QR code
        const parser = new DOMParser();
        const svgDoc = parser.parseFromString(data.qrCode, 'image/svg+xml');
        const svgElement = svgDoc.documentElement;

        // Clear previous QR code if any
        const qrCodeContainer = document.getElementById('qrCode');
        qrCodeContainer.innerHTML = '';
        qrCodeContainer.appendChild(svgElement);

        // Show the QR container
        document.getElementById('qrContainer').classList.remove('hidden');
        
        console.log('QR code displayed successfully');
    } catch (error) {
        console.error('Error:', error);
        alert(`Error: ${error.message}`);
    }
});
